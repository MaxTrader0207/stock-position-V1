async function analyzeStock() { const stockList = document.getElementById("stockInput").value; const btn = document.getElementById("analyzeBtn"); const resultArea = document.getElementById("resultArea"); const analysisText = document.getElementById("analysisText"); const audioPlayer = document.getElementById("audioPlayer"); if (!stockList.trim()) return alert("請輸入庫存資料"); btn.disabled = true; btn.innerText = "分析與生成語音中..."; resultArea.style.display = "none"; try { const response = await fetch(WORKER_URL, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ stockList }) }); if (!response.ok) { const errData = await response.json(); throw new Error(errData.error || "伺服器發生錯誤"); } // 1. 從 Header 讀取分析文字並解碼 const encodedText = response.headers.get("X-Analysis-Text"); if (encodedText) { analysisText.innerText = decodeURIComponent(encodedText); } // 2. 直接將音檔 Response 轉成 Blob 播放（極速且不會爆記憶體） const audioBlob = await response.blob(); audioPlayer.src = URL.createObjectURL(audioBlob); resultArea.style.display = "block"; audioPlayer.play(); } catch (err) { alert("發生錯誤：" + err.message); } finally { btn.disabled = false; btn.innerText = "開始分析與語音播報"; } }
